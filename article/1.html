不同的编程语言输出Hello World
1.Python 
print("Hello World!")

2.C语言 
#include<stdio.h>
void main()
{
    printf("Hello World!");
    return 0;
}

3.C++ 
#include<iostream>
using namespace std;
int main()
{
    cout<<"Hello World!";
    return 0;
}

4.Bash 
#!usr/bin/sh 
echo "Hello World!";

5.Ruby 
puts "Hello World";

6.PHP 
<?php
    echo "Hello World!";
?>

7.Perl 
#!usr/bin/perl
print("Hello World!");

8.汇编语言 
;以下代码来源于百度百科
DATA   SEGMENT                            
PRINT  DB "Hello World!", 0AH, 0DH, '$'  
DATA   ENDS                  
  
STACK  SEGMENT   STACK           
       DW  20  DUP(0)                       
STACK  ENDS                  
  
ASSUME CS:CODE, DS:DATA, SS:STACK            
  
CODE   SEGMENT                               
START:                                       
        MOV AX, DATA                         
        MOV DS, AX                           
        MOV DX, OFFSET  PRINT                
        MOV AH, 09H
        INT 21H                              
        MOV AH, 4CH                          
        INT 21H
CODE   ENDS                                  
END    START
